1. 查看用户/用户组信息

$$ echo $USER                             #显示当前用户

$ cat /etc/passwd                        #查看所有用户信息

$ cat /etc/shadow                        #查看所有用户的加密信息

$ cat /etc/group                          #查看所有用户组信息

2. 用户操作

2.1 创建用户

$ useradd 用户名                                             #创建用户

$ useradd -u UID 用户名                                  #创建用户同时指定UID

$ useradd -u UID -g GID 用户名                       #创建用户同时指定UID和GID

$ passed 用户名                                               #设置密码

2.2 修改用户信息

$usermod -g 修改后的GID 用户名                    #修改用户所属的组

$usermod -c “” 用户名                                    #修改备注信息

2.3 删除用户

$ userdel -r 用户名                                         #删除用户，如果省略-r则不能彻底删除


3. 用户组操作

3.1 创建用户组

$ groupadd 组名                                            #创建用户组

3.2 修改组ID

$ groupmod -g 修改后的GID 组名                   #修改组ID

3.3 删除组

$ groupdel 组名                                              #删除组


4. 文件拥有者

4.1 查看文件拥有者，文件所属用户组

4.2 修改文件拥有者和文件所属用户组

$ chown 修改后的用户名 filename                  #修改文件的拥有者

$ chown 修改后的用户名:修改后的组 filename #同时修改文件拥有者和文件所属用户组

$ chgrp 修改后的组 filename                          #修改文件所属的用户组
